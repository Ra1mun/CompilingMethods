ВЕЩ a[n];

ВЫВОД("Введите элементы массива:");

ПОКА (i < n) {
    ВВОД(a[i]);
    i = i + 1;
}

i = 0;
ВЕЩ j;
ВЕЩ temp;

ПОКА (i < n - 1) {
    j = 0;
    ПОКА (j < n - i - 1) {
        ЕСЛИ (a[j] > a[j+1]) {
            temp = a[j];
            a[j] = a[j+1];
            a[j+1] = temp;
        }
        j = j + 1;
    }
    i = i + 1;
}
